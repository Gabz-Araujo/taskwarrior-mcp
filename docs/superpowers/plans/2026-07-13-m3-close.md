**Tech Stack:** TypeScript (NodeNext, strict + `exactOptionalPropertyTypes`), Zod v4, `@modelcontextprotocol/sdk` 1.29, Node 24 built-in test runner (`node:test`) run through `tsx` for TS execution.

```ts
import type { Task, ListFilter, ListSort } from "./types.js";

export function buildListFilterArgs(filter?: ListFilter): string[] {
  const args: string[] = [];
  const status = filter?.status ?? "pending";
  if (status !== "all") args.push(`status:${status}`);
  if (filter?.project) args.push(`project:${filter.project}`);
  for (const tag of filter?.tags ?? []) args.push(`+${tag}`);
  if (filter?.dueBefore) args.push(`due.before:${filter.dueBefore}`);
  if (filter?.dueAfter) args.push(`due.after:${filter.dueAfter}`);
  return args;
}

function cmpAsc(a: string, b: string): number {
  return a < b ? -1 : a > b ? 1 : 0;
}

function cmpOptionalAsc(a: string | undefined, b: string | undefined): number {
  if (a === undefined && b === undefined) return 0;
  if (a === undefined) return 1;
  if (b === undefined) return -1;
  return cmpAsc(a, b);
}

export function sortTasks(tasks: Task[], sort?: ListSort): Task[] {
  if (!sort) return tasks;
  const copy = [...tasks];
  if (sort === "urgency") {
    copy.sort((a, b) => (b.urgency ?? 0) - (a.urgency ?? 0));
  } else if (sort === "due") {
    copy.sort((a, b) => cmpOptionalAsc(a.due, b.due));
  } else {
    copy.sort((a, b) => cmpAsc(a.entry, b.entry));
  }
  return copy;
}
```

```ts
import type {
  Taskwarrior,
  Task,
  AddOptions,
  ListFilter,
  ListOptions,
  ModifyOptions,
} from "../taskwarrior/index.js";
import { TaskwarriorError } from "../taskwarrior/index.js";
import { sortTasks } from "../taskwarrior/query.js";

const NOT_FOUND = (uuid: string) =>
  new TaskwarriorError(
    `No task matches uuid ${uuid} — call list_tasks or get_task to find valid uuids.`,
  );

export class FakeTaskwarrior implements Taskwarrior {
  private tasks = new Map<string, Task>();
  private counter = 0;

  private nextUuid(): string {
    this.counter += 1;
    return `00000000-0000-0000-0000-${String(this.counter).padStart(12, "0")}`;
  }

  async add(description: string, options?: AddOptions): Promise<Task> {
    const uuid = this.nextUuid();
    const task: Task = {
      uuid,
      description,
      status: "pending",
      entry: `20260101T0000${String(this.counter).padStart(2, "0")}Z`,
      urgency: this.counter,
      ...(options?.project !== undefined ? { project: options.project } : {}),
      ...(options?.due !== undefined ? { due: options.due } : {}),
      ...(options?.priority !== undefined ? { priority: options.priority } : {}),
      ...(options?.tags !== undefined ? { tags: options.tags } : {}),
    };
    this.tasks.set(uuid, task);
    return task;
  }

  async list(filter?: ListFilter, options?: ListOptions): Promise<Task[]> {
    const status = filter?.status ?? "pending";
    let out = [...this.tasks.values()];
    if (status !== "all") out = out.filter((t) => t.status === status);
    if (filter?.project) out = out.filter((t) => t.project === filter.project);
    if (filter?.tags?.length) {
      const wanted = filter.tags;
      out = out.filter((t) => wanted.every((tag) => t.tags?.includes(tag)));
    }
    out = sortTasks(out, options?.sort);
    return options?.limit !== undefined ? out.slice(0, options.limit) : out;
  }

  async modify(uuid: string, options: ModifyOptions): Promise<Task> {
    const task = this.tasks.get(uuid);
    if (!task) throw NOT_FOUND(uuid);
    const updated: Task = {
      ...task,
      ...(options.description !== undefined ? { description: options.description } : {}),
      ...(options.project !== undefined ? { project: options.project } : {}),
      ...(options.due !== undefined ? { due: options.due } : {}),
      ...(options.priority !== undefined ? { priority: options.priority } : {}),
    };
    this.tasks.set(uuid, updated);
    return updated;
  }

  async done(uuid: string): Promise<Task> {
    const task = this.tasks.get(uuid);
    if (!task) throw NOT_FOUND(uuid);
    const updated: Task = { ...task, status: "completed" };
    this.tasks.set(uuid, updated);
    return updated;
  }

  async delete(uuid: string): Promise<Task> {
    const task = this.tasks.get(uuid);
    if (!task) throw NOT_FOUND(uuid);
    const updated: Task = { ...task, status: "deleted" };
    this.tasks.set(uuid, updated);
    return updated;
  }

  async getByUuid(uuid: string): Promise<Task | undefined> {
    return this.tasks.get(uuid);
  }
}
```

```ts
function compact<T extends object>(obj: T): { [K in keyof T]: Exclude<T[K], undefined> } {
  const out: Record<string, unknown> = {};
  for (const [key, value] of Object.entries(obj)) {
    if (value !== undefined) out[key] = value;
  }
  return out as { [K in keyof T]: Exclude<T[K], undefined> };
}

function ok(structured: Record<string, unknown>): CallToolResult {
  return {
    content: [{ type: "text", text: JSON.stringify(structured, null, 2) }],
    structuredContent: structured,
  };
}

function fail(message: string): CallToolResult {
  return { isError: true, content: [{ type: "text", text: message }] };
}

async function guard<T>(
  fn: () => Promise<T>,
  shape: (result: T) => Record<string, unknown>,
): Promise<CallToolResult> {
  try {
    return ok(shape(await fn()));
  } catch (error) {
    if (error instanceof TaskwarriorError) return fail(error.message);
    throw error;
  }
}
```
